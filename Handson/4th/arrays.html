Question 9: Unique Color Shirt?

Ans.

                function Unique_Shirts(arr, N) {

                    let uniqueElement = 0;
        
                    for (let i = 0; i < N; i++) {
                        let count = 0, key = arr[i]; //count & key will initialize 
                        // with 0 and i value  when loop will repeat. 
        
                        // we have to use second loop b'cause every time each
                        // no we have to check with each no which is present in
                        // the array.
                        for (let j = 0; j < N; j++) {     // 0 to 5 (6 times)
                            if (key == arr[j])//it'll match number and if matches
                            {
                                count++;     //count will increases by one;
                            }
                        }
                        if (count == 1) {
                            uniqueElement++;//it'll increase when element is
                            // available in array only 1 time.
                        }
                    }
                    return uniqueElement;
                }

Question 10: Min & Max?

Ans.

                function arrayMin(arr) {
                    let min=arr[0];
                    for(let i=0; i< arr.length; i++){
                      if(arr[i]< min){
                        min=arr[i];
                      }
                    }
                    return min;
                }
                  
                function arrayMax(arr) {
                   let max=arr[0];
                   for(let i=0; i< arr.length; i++){
                     if(arr[i]>max){
                       max=arr[i];
                     }
                   }
                   return max;
                }

Question 11: Birthday Game

Ans.

                function Birthday_Game(arr,D ,M) {
                    let count=0;
                    for(let i=0; i <= arr.length-M; i++){
                      let sum = 0;
                      for(let j=0; j< M; j++){
                        sum += arr[i+j];
                      }
                      if(sum == D){
                        count++;
                      }
                    }
                    return count;
                }
