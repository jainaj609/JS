                let count = 0, check = 0;

                    check = N % 10;

Question 7: Check whether a Number is a prime or not?

Ans.

            const Prime_Check = (N) => {
                let s = "";
                if (N == 1) {
                    s = "NO";
                }
                for (let i = 2; i < N; i++) {
                    if (N % i == 0) {
                        s = "NO";
                    }
                }
                if (s == "NO") {
                    return "NO";
                } else {
                    return "YES";
                }
            };
